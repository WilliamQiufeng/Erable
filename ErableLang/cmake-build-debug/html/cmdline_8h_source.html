366  bool parse(const std::string &arg) {
367  std::vector<std::string> args;
368 
369  std::string buf;
370  bool in_quote = false;
371  for (std::string::size_type i = 0; i < arg.length(); i++) {
372  if (arg[i] == '\"') {
373  in_quote = !in_quote;
374  continue;
375  }
376 
377  if (arg[i] == ' ' && !in_quote) {
378  args.push_back(buf);
379  buf = "";
380  continue;
381  }
382 
383  if (arg[i] == '\\') {
384  i++;
385  if (i >= arg.length()) {
386  errors.push_back("unexpected occurrence of '\\' at end of string");
387  return false;
388  }
389  }
390 
391  buf += arg[i];
392  }
393 
394  if (in_quote) {
395  errors.push_back("quote is not closed");
396  return false;
397  }
398 
399  if (buf.length() > 0)
400  args.push_back(buf);
401 
402  for (size_t i = 0; i < args.size(); i++)
403  std::cout << "\"" << args[i] << "\"" << std::endl;
404 
405  return parse(args);
406  }

418  bool parse(int argc, const char *const argv[]) {
419  errors.clear();
420  others.clear();
421 
422  if (argc < 1) {
423  errors.push_back("argument number must be longer than 0");
424  return false;
425  }
426  if (prog_name == "")
427  prog_name = argv[0];
428 
429  std::map<char, std::string> lookup;
430  for (std::map<std::string, option_base *>::iterator p = options.begin();
431  p != options.end(); p++) {
432  if (p->first.length() == 0) continue;
433  char initial = p->second->short_name();
434  if (initial) {
435  if (lookup.count(initial) > 0) {
436  lookup[initial] = "";
437  errors.push_back(std::string("short option '") + initial + "' is ambiguous");
438  return false;
439  } else lookup[initial] = p->first;
440  }
441  }
442 
443  for (int i = 1; i < argc; i++) {
444  if (strncmp(argv[i], "--", 2) == 0) {
445  const char *p = strchr(argv[i] + 2, '=');
446  if (p) {
447  std::string name(argv[i] + 2, p);
448  std::string val(p + 1);
449  set_option(name, val);
450  } else {
451  std::string name(argv[i] + 2);
452  if (options.count(name) == 0) {
453  errors.push_back("undefined option: --" + name);
454  continue;
455  }
456  if (options[name]->has_value()) {
457  if (i + 1 >= argc) {
458  errors.push_back("option needs value: --" + name);
459  continue;
460  } else {
461  i++;
462  set_option(name, argv[i]);
463  }
464  } else {
465  set_option(name);
466  }
467  }
468  } else if (strncmp(argv[i], "-", 1) == 0) {
469  if (!argv[i][1]) continue;
470  char last = argv[i][1];
471  for (int j = 2; argv[i][j]; j++) {
472  last = argv[i][j];
473  if (lookup.count(argv[i][j - 1]) == 0) {
474  errors.push_back(std::string("undefined short option: -") + argv[i][j - 1]);
475  continue;
476  }
477  if (lookup[argv[i][j - 1]] == "") {
478  errors.push_back(std::string("ambiguous short option: -") + argv[i][j - 1]);
479  continue;
480  }
481  set_option(lookup[argv[i][j - 1]]);
482  }
483 
484  if (lookup.count(last) == 0) {
485  errors.push_back(std::string("undefined short option: -") + last);
486  continue;
487  }
488  if (lookup[last] == "") {
489  errors.push_back(std::string("ambiguous short option: -") + last);
490  continue;
491  }
492 
493  if (i + 1 < argc && options[lookup[last]]->has_value()) {
494  set_option(lookup[last], argv[i + 1]);
495  i++;
496  } else {
497  set_option(lookup[last]);
498  }
499  } else {
500  others.push_back(argv[i]);
501  }
502  }
503 
504  for (std::map<std::string, option_base *>::iterator p = options.begin();
505  p != options.end(); p++)
506  if (!p->second->valid())
507  errors.push_back("need option: --" + std::string(p->first));
508 
509  return errors.size() == 0;
510  }
